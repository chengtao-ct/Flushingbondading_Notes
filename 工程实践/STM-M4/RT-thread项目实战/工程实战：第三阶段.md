服务器核心循环
static void web_server_loop(void)
{
    int server;                         // 监听 socket fd
    int client;                         // 客户端连接 socket fd
    int opt = 1;
    struct sockaddr_in addr;            // 服务器地址
    struct sockaddr_in client_addr;     // 客户端地址
    socklen_t client_len;
第一步：创建 socket
    server = socket(AF_INET, SOCK_STREAM, 0);
    // AF_INET     = IPv4
    // SOCK_STREAM = TCP（可靠传输）
    // 0           = 默认协议（TCP）
    // 返回值 > 0 是文件描述符，< 0 是失败
第二步：设置地址复用
    setsockopt(server, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));
作用：服务器重启时能立即绑定端口，否则要等 2~4 分钟的 TIME_WAIT 超时。嵌入式设备经常重启，这行很重要。
第三步：绑定地址
    memset(&addr, 0, sizeof(addr));
    addr.sin_family = AF_INET;              // IPv4
    addr.sin_port = htons(WEB_SERVER_PORT); // 端口 80，htons 转网络字节序（大端）
    addr.sin_addr.s_addr = htonl(INADDR_ANY);  // 0.0.0.0，监听所有网卡
    if (bind(server, (struct sockaddr *)&addr, sizeof(addr)) < 0)
    {
        rt_kprintf("[web] bind failed\r\n");
        closesocket(server);                // 失败必须关闭 socket，否则 fd 泄漏
        return;
    }
第四步：开始监听
    if (listen(server, WEB_LISTEN_BACKLOG) < 0)
    // WEB_LISTEN_BACKLOG = 等待队列长度
    // 比如设为 5：同时可以有 5 个连接在排队等 accept
    {
        rt_kprintf("[web] listen failed\r\n");
        closesocket(server);
        return;
    }
    rt_kprintf("[web] listen on port %d\r\n", WEB_SERVER_PORT);
第五步：接受连接（死循环）
    while (1)
    {
        client_len = sizeof(client_addr);
        client = accept(server, (struct sockaddr *)&client_addr, &client_len);
        // accept() 阻塞等待，直到有客户端连接
        // 返回新的 socket fd 专门用于和这个客户端通信
        if (client < 0)
        {
            rt_kprintf("[web] accept failed\r\n");
            continue;                       // 失败了继续等下一个连接
        }
        rt_kprintf("[web] client connected\r\n");
        closesocket(client);                // ★ 目前直接关闭！还没实现 HTTP 解析
    }
}

## Key Changes

- 先整理当前 `web_service.c` 骨架：
    
    - 删除重复的未加 `#ifndef WEB_SERVICE_HOST_TEST` 的 `web_thread_entry` / `web_server_loop` 声明。
    - 保留 `WEB_SERVICE_HOST_TEST` 分支，使纯函数可在 PC host test 下编译。
    - 保持 web 线程优先级 `24`、栈 `4096`、端口 `80` 不变。
- 新增 HTTP 处理闭环：
    
    - `web_server_loop()` 中 `accept()` 后不再直接关闭 client，而是调用 `web_handle_client(client)`。
    - `web_handle_client()` 负责 `recv()` 请求、解析请求行、提取 method/path/body、路由到具体 API。
    - 每个连接只处理一个 HTTP 请求，响应后立即 `closesocket(client)`。
- 实现响应接口：
    
    - `GET /` 返回 `200 text/plain`，内容类似 `industrial gateway api`.
    - `GET /api/status` 调用 `igw_model_get_snapshot()`，再用 `web_service_build_status_json()` 返回 JSON。
    - `POST /api/control` 使用 `web_service_parse_control_form()` 校验参数，然后按存在字段调用 `igw_model_set_mode()`、`igw_model_set_target()`、`igw_model_set_manual_pwm()`，最后返回最新状态 JSON。
    - 非法参数返回 `400 Bad Request`；未知路径 `404`；不支持的方法 `405`；模型读取失败 `503`；响应缓冲不足 `500`。
- HTTP v1 约束：
    
    - 请求缓冲区 `1024`，响应缓冲区 `1024`。
    - `POST /api/control` 只接受简单 ASCII 表单，不做 URL decode。
    - 不开放蜂鸣器、报警阈值、校准等高风险控制项。
    - 不做认证、不做公网安全，只用于局域网调试。
