#include <stdbool.h>
#include <stdlib.h>

typedef struct {
    int *queue;
    int head;
    int tail;
    int count;
    int capacity;
} MyCircularQueue;

// 1. 初始化 (双重 malloc)
MyCircularQueue* myCircularQueueCreate(int k) {
    MyCircularQueue* obj = (MyCircularQueue*)malloc(sizeof(MyCircularQueue));
    obj->queue = (int*)malloc(k * sizeof(int));
    obj->capacity = k;
    obj->head = 0;
    obj->tail = 0;
    obj->count = 0;
    return obj;
}

// 2. 入队
bool myCircularQueueEnQueue(MyCircularQueue* obj, int value) {
    if (obj->count == obj->capacity) { // 判满
        return false;
    }
    obj->queue[obj->tail] = value;
    obj->tail = (obj->tail + 1) % obj->capacity; // 绕圈
    obj->count++;
    return true;
}

// 3. 出队
bool myCircularQueueDeQueue(MyCircularQueue* obj) {
    if (obj->count == 0) { // 判空
        return false;
    }
    obj->head = (obj->head + 1) % obj->capacity; // 绕圈
    obj->count--;
    return true;
}

// 4. 获取队首元素
int myCircularQueueFront(MyCircularQueue* obj) {
    if (obj->count == 0) {
        return -1; // LeetCode 要求为空时返回 -1
    }
    return obj->queue[obj->head]; // head 直接指向的就是队首
}

// 5. 获取队尾元素
int myCircularQueueRear(MyCircularQueue* obj) {
    if (obj->count == 0) {
        return -1;
    }
    // 💡 极客魔法：防负数取模
    int real_tail_index = (obj->tail - 1 + obj->capacity) % obj->capacity;
    return obj->queue[real_tail_index];
}

// 6. 判空
bool myCircularQueueIsEmpty(MyCircularQueue* obj) {
    return obj->count == 0;
}

// 7. 判满
bool myCircularQueueIsFull(MyCircularQueue* obj) {
    return obj->count == obj->capacity;
}

// 8. 销毁队列 (打扫战场)
void myCircularQueueFree(MyCircularQueue* obj) {
    free(obj->queue); // 先释放内部的数组
    free(obj);        // 再释放结构体本身
}
