#include <stdbool.h>
#include <string.h>
#include <stdlib.h>

bool isValid(char * s) {
    int len = strlen(s);
    
    // 💡 极客优化 1：奇数长度直接判死刑
    // 如果报文长度是奇数，绝对不可能两两配对，直接拦截，节省 CPU 时间
    if (len % 2 != 0) {
        return false;
    }

    // 💡 极客优化 2：堆内存申请 vs 栈内存申请
    // 很多新手会写：char stack[len]; 
    // 这在单片机中是致命的！如果 len 是 10000，这会瞬间吃掉 10KB 的系统栈（Stack），导致单片机 HardFault 宕机。
    // 所以，我们老老实实去堆（Heap）里申请这块内存。
    char* stack = (char*)malloc(len * sizeof(char));
    
    // 初始化栈顶游标。-1 代表目前栈里空无一物
    int top = -1; 

    // 开始逐个字节解析字符串
    for (int i = 0; i < len; i++) {
        char ch = s[i];
        
        // 状态 A：如果是左括号，无脑推进栈里 (Push)
        if (ch == '(' || ch == '[' || ch == '{') {
            top++;              // 游标先往上走一格
            stack[top] = ch;    // 把字符放进这个格子里
            // 这两句可以缩写为：stack[++top] = ch;
        } 
        // 状态 B：如果是右括号，准备消消乐 (Pop)
        else {
            // 🚨 错误拦截 1：右括号来了，但栈是空的！
            // 比如报文是 "]" 或者 "())"，根本没有左括号等着匹配
            if (top == -1) {
                free(stack); // 走之前千万别忘了释放内存！
                return false;
            }
            
            // 把栈顶的元素拿出来看看
            char topChar = stack[top];
            
            // 检查：来的右括号，和栈顶的左括号，是不是原配？
            if ((ch == ')' && topChar == '(') ||
                (ch == ']' && topChar == '[') ||
                (ch == '}' && topChar == '{')) {
                // 匹配成功！执行出栈 (Pop)
                // 注意：在 C 语言的底层，我们不需要去“清空”原来那个位置的字符，
                // 只需要把游标往下移一格，下次再 Push 就会直接覆盖它。
                top--; 
            } else {
                // 🚨 错误拦截 2：类型不匹配！
                // 比如报文是 "(]"，栈顶是 '('，来的却是 ']'
                free(stack); // 走之前释放内存！
                return false;
            }
        }
    }

    // 状态 C：字符串全看完了，来做最终结算
    // 如果完全匹配，栈里应该空无一物（top 必须回到了 -1）。
    // 如果 top > -1，说明还有孤零零的左括号留在栈里没被闭合（比如报文是 "((" ）
    bool result = (top == -1);
    
    // 💡 极客优化 3：打扫战场
    // 自己 malloc 的内存，含着泪也要自己 free 掉，这是嵌入式工程师的职业操守
    free(stack);
    
    return result;
}
