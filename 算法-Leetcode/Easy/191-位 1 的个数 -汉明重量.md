```c
int hammingWeight(uint32_t n) {
    int count = 0;          // 准备一个计数器
    
    while (n != 0) {        // 只要 n 还没被榨干成 0
        n = n & (n - 1);    // 扣动扳机：抹掉最右边的一个 1
        count++;            // 击杀数 + 1
    }
    
    return count;           // 完工，汇报击杀数！
}
```
