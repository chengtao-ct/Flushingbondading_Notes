#### 第一步：安顿新人的左手

新人 `B` 刚来，第一件事是赶紧抱住老大 `Head` 的大腿。

- **动作：** 让 `B` 的 `prev` 指针指向 `Head`。
    
- **C语言：** `node->prev = head;`
    

#### 第二步：安顿新人的右手

新人 `B` 还要拉住原来排第一的 `A`。注意，此时 `Head` 的右手依然还牵着 `A`，所以我们可以通过 `head->next` 找到 `A`。

- **动作：** 让 `B` 的 `next` 指针指向 `A` (`head->next`)。
    
- **C语言：** `node->next = head->next;`
    
- _(此时，新人 `B` 已经稳稳地抓住了两边，但旧的连线还没断。)_
    

#### 第三步：原来老二（A）的左手换人

`A` 发现新人 `B` 已经到位了，于是 `A` 松开抓着老大 `Head` 的左手，改去抓新人 `B`。

- **动作：** 找到 `A` (`head->next`)，把它的 `prev` 指针改指给 `node`。
    
- **C语言：** `head->next->prev = node;`
    
- **⚠️ 极度危险警告：** 这一步必须在第四步之前做！如果你先把 `head->next` 改成了别人，你就永远找不到 `A` 了！
    

#### 第四步：老大（Head）的右手换人

既然小弟们都已经牵好手了，老大 `Head` 终于可以松开原来抓着 `A` 的右手，改去牵新人 `B` 了。

- **动作：** 让 `Head` 的 `next` 指针指向 `node`。
    
- **C语言：** `head->next = node;`
    

**最终状态：** `[Head] <===> [B] <===> [A]`

老兵的终极秘籍是：**永远在链表两端放两个绝对不会被删除的“哨兵”：虚拟 Head 和虚拟 Tail。** 初始状态：`Head <==> Tail`




```c
#include <stdlib.h>

// ==========================================
// 1. 底层数据结构定义
// ==========================================

// 双向链表节点
typedef struct DLinkedNode {
    int key;    // 必须存 key，因为淘汰时要拿着它去哈希表销户
    int value;
    struct DLinkedNode* prev;
    struct DLinkedNode* next;
} DLinkedNode;

// LRU 缓存主机
typedef struct {
    int capacity;                 // 最大容量
    int size;                     // 当前数据量
    DLinkedNode* head;            // 虚拟头节点 (哨兵，代表最新)
    DLinkedNode* tail;            // 虚拟尾节点 (哨兵，代表最老)
    DLinkedNode* hash[10001];     // 哈希表：下标是 key，存的是节点指针 (题目限定 key <= 10000)
} LRUCache;

// ==========================================
// 2. 底层四步握手零件 (双向链表操作)
// ==========================================

// 零件 A：将节点安插到虚拟老大 (Head) 的正后方
void addNode(DLinkedNode* head, DLinkedNode* node) {
    node->prev = head;
    node->next = head->next;
    head->next->prev = node;
    head->next = node;
}

// 零件 B：无情摘除某个节点 (不管它在哪)
void removeNode(DLinkedNode* node) {
    node->prev->next = node->next;
    node->next->prev = node->prev;
}

// 零件 C：刷新存在感 (先摘除，再插到最前面)
void moveToHead(DLinkedNode* head, DLinkedNode* node) {
    removeNode(node);
    addNode(head, node);
}

// 零件 D：淘汰尾部倒霉蛋，并把它揪出来返回
DLinkedNode* removeTail(DLinkedNode* tail) {
    DLinkedNode* evicted = tail->prev; // 拿到实体
    removeNode(evicted);               // 摘除连线
    return evicted;                    // 返回给上层销毁
}

// ==========================================
// 3. 业务主控逻辑
// ==========================================

// 系统启动：初始化缓存主机
LRUCache* lRUCacheCreate(int capacity) {
    LRUCache* obj = (LRUCache*)malloc(sizeof(LRUCache));
    obj->capacity = capacity;
    obj->size = 0;

    // 创建虚拟头尾哨兵
    obj->head = (DLinkedNode*)malloc(sizeof(DLinkedNode));
    obj->tail = (DLinkedNode*)malloc(sizeof(DLinkedNode));
    
    // 哨兵牵手：Head <==> Tail
    obj->head->prev = NULL;
    obj->head->next = obj->tail;
    obj->tail->prev = obj->head;
    obj->tail->next = NULL;

    // 清空哈希表
    for(int i = 0; i < 10001; i++) {
        obj->hash[i] = NULL;
    }
    return obj;
}

// 读取操作：光速查找 + 刷新活跃度
int lRUCacheGet(LRUCache* obj, int key) {
    DLinkedNode* node = obj->hash[key]; // 从哈希表瞬间找到内存地址
    
    if (node == NULL) {
        return -1; // 没查到
    }
    
    // 查到了！提拔到最前面
    moveToHead(obj->head, node);
    return node->value;
}

// 写入操作：更新/新建 + 满载淘汰
void lRUCachePut(LRUCache* obj, int key, int value) {
    DLinkedNode* node = obj->hash[key];

    // 场景 1：数据已存在 -> 更新值，刷新活跃度
    if (node != NULL) {
        node->value = value;
        moveToHead(obj->head, node);
        return;
    }

    // 场景 2：全新数据 -> 造新兵，入列
    DLinkedNode* newNode = (DLinkedNode*)malloc(sizeof(DLinkedNode));
    newNode->key = key;
    newNode->value = value;
    
    obj->hash[key] = newNode;       // 哈希表登记
    addNode(obj->head, newNode);    // 插到链表最前面
    obj->size++;

    // 场景 3：容量警报 -> 触发无情淘汰
    if (obj->size > obj->capacity) {
        DLinkedNode* tailNode = removeTail(obj->tail); // 揪出最老的数据
        obj->hash[tailNode->key] = NULL;               // 哈希表销户
        free(tailNode);                                // 内存火化
        obj->size--;
    }
}

// ==========================================
// 4. 收尾工作：释放整个系统的内存 (防内存泄漏)
// ==========================================
void lRUCacheFree(LRUCache* obj) {
    DLinkedNode* curr = obj->head;
    // 顺藤摸瓜，干掉所有节点
    while (curr != NULL) {
        DLinkedNode* temp = curr->next;
        free(curr);
        curr = temp;
    }
    // 最后砸掉主机
    free(obj);
}

 * int param_1 = lRUCacheGet(obj, key);
 * lRUCachePut(obj, key, value);
 * lRUCacheFree(obj);
 */
```
