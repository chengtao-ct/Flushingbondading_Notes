> [!abstract] 核心本质
> 异构SoC常包含多种物理内存（内部SRAM快而小、外部SDRAM慢而大）。`[[heap_5]]` 方案允许将多个**非连续物理区域**纳入统一分配框架，实现"逻辑上单一堆、物理上分区管理"。

## 一、核心定义

**异构内存架构**：现代SoC典型的内存布局包含多个物理特性不同的区域：

```
SoC 内存地址空间
┌─────────────────────────────────────────────────────┐ 0x20000000
│  内部 SRAM (128KB)                                  │
│  - 零等待状态，最高速                               │
│  - 紧耦合总线，适合实时控制                         │
├─────────────────────────────────────────────────────┤ 0x60000000
│  外部 SDRAM/PSRAM (8MB~64MB)                        │
│  - 有等待状态，延迟较高                             │
│  - 适合大数据缓冲、日志、图像                       │
├─────────────────────────────────────────────────────┤ 0x24000000
│  SRAM2/Backup SRAM (可选)                           │
│  - 可配置为DMA专用区                                │
│  - 低功耗模式下保持                                 │
└─────────────────────────────────────────────────────┘
```

**heap_5 解决的核心问题**：
- 将多个物理上不连续的内存区域"拼接"成一个逻辑堆
- 分配时自动选择合适的区域
- 支持运行时动态添加内存区域

## 二、底层原理

### 2.1 FreeRTOS Heap方案对比

| 方案 | 内存区域 | 碎片处理 | 适用场景 | 时间确定性 |
|------|----------|----------|----------|------------|
| heap_1 | 单一连续区 | 无碎片（只分配不释放） | 静态任务模型 | ✅ 确定 |
| heap_2 | 单一连续区 | 有碎片（最佳匹配） | 频繁分配相同大小 | ⚠️ 不确定 |
| heap_3 | 单一连续区 | 依赖标准库 | 简单应用 | ❌ 不确定 |
| heap_4 | 单一连续区 | 合并相邻空闲块 | 通用场景 | ⚠️ 较确定 |
| **heap_5** | **多非连续区** | 合并相邻空闲块 | **异构内存** | ⚠️ 较确定 |

### 2.2 heap_5 核心数据结构

```c
// heap_5.c 核心结构
typedef struct HeapRegion
{
    uint8_t *pucStartAddress;  // 区域起始地址
    size_t xSizeInBytes;       // 区域大小（字节）
} HeapRegion_t;

// 内存块头部（隐藏实现）
typedef struct A_BLOCK_LINK
{
    struct A_BLOCK_LINK *pxNextFreeBlock;  // 下一个空闲块
    size_t xBlockSize;                      // 块大小（含头部）
} BlockLink_t;

// 关键：空闲块链表按地址排序，支持跨区域合并
```

### 2.3 heap_5 初始化流程

```c
// 1. 定义内存区域表（必须按地址升序排列！）
static const HeapRegion_t xHeapRegions[] =
{
    { (uint8_t *)0x20000000, 0x00020000 },  // 内部SRAM: 128KB @ 0x20000000
    { (uint8_t *)0x60000000, 0x00800000 },  // 外部SDRAM: 8MB @ 0x60000000
    { NULL, 0 }                              // 结束标记
};

// 2. 在调度器启动前初始化
void vApplicationDaemonTaskStartupHook(void)
{
    // 必须在创建任何任务/队列之前调用
    vPortDefineHeapRegions(xHeapRegions);
    
    printf("Heap initialized: Total %u bytes\n", 
           xPortGetFreeHeapSize());
}

// 3. 分配时自动选择区域
void *ptr = pvPortMalloc(1024);  // 可能来自SRAM或SDRAM
```

> [!warning] 关键约束
> `xHeapRegions[]` **必须按起始地址升序排列**！否则空闲块链表会混乱，导致内存破坏。

### 2.4 分配策略

heap_4/heap_5 采用 **首次适应算法**：

```
空闲链表: [Block1@0x20000000] -> [Block2@0x20008000] -> [Block3@0x60000000]

请求 2KB:
1. 从链表头开始遍历
2. 找到第一个 >= 2KB 的块
3. 如果块远大于需求，分裂为两块
4. 返回用户可用地址（块头之后）
```

**问题**：小对象可能被分配到外部慢内存！

## 三、实战应用落地

### 3.1 分区策略设计

```c
// memory_policy.h - 分区策略定义

// 内存区域枚举
typedef enum
{
    MEM_REGION_FAST = 0,    // 内部SRAM - 实时控制
    MEM_REGION_BULK,        // 外部SDRAM - 大数据
    MEM_REGION_DMA,         // DMA专用区（无Cache或Cache一致性）
    MEM_REGION_COUNT
} MemoryRegion_e;

// 对象类型到区域的映射
typedef struct
{
    const char *name;
    MemoryRegion_e preferredRegion;
    size_t typicalSize;
    bool requireDmaCapable;
} MemoryPolicy_t;

// 策略表
static const MemoryPolicy_t xMemoryPolicies[] =
{
    { "TCB",           MEM_REGION_FAST,  sizeof(TCB_t),      false },
    { "TaskStack",     MEM_REGION_FAST,  2048,               false },
    { "QueueBuffer",   MEM_REGION_FAST,  128,                false },
    { "ImageBuffer",   MEM_REGION_BULK,  640*480*2,          true  },
    { "LogBuffer",     MEM_REGION_BULK,  4096,               false },
    { "DmaDescriptor", MEM_REGION_DMA,   32,                 true  },
};
```

### 3.2 封装API实现

```c
// memory_api.c - 分区封装

#include "FreeRTOS.h"
#include "task.h"

// 区域边界（从链接脚本获取）
extern uint32_t __SRAM_START__;
extern uint32_t __SRAM_END__;
extern uint32_t __SDRAM_START__;
extern uint32_t __SDRAM_END__;

// 判断指针属于哪个区域
static MemoryRegion_e prvGetPointerRegion(void *ptr)
{
    uint32_t addr = (uint32_t)ptr;
    
    if (addr >= (uint32_t)&__SRAM_START__ && 
        addr < (uint32_t)&__SRAM_END__) {
        return MEM_REGION_FAST;
    }
    if (addr >= (uint32_t)&__SDRAM_START__ && 
        addr < (uint32_t)&__SDRAM_END__) {
        return MEM_REGION_BULK;
    }
    return MEM_REGION_COUNT;  // 未知区域
}

// 快速内存分配（强制内部SRAM）
void *pvAllocFast(size_t xSize)
{
    void *ptr = pvPortMalloc(xSize);
    
    // 验证分配结果
    if (ptr != NULL && prvGetPointerRegion(ptr) != MEM_REGION_FAST) {
        // 分配到了外部内存，这是策略违规！
        // 可以选择：释放重试、记录警告、或触发断言
        vPortFree(ptr);
        return NULL;  // 或尝试其他策略
    }
    
    return ptr;
}

// 大块内存分配（优先外部SDRAM）
void *pvAllocBulk(size_t xSize)
{
    // heap_5 会优先使用低地址区域
    // 对于大块数据，可能需要反向策略
    
    void *ptr = pvPortMalloc(xSize);
    return ptr;  // 接受任何区域
}

// DMA安全内存分配
void *pvAllocDmaSafe(size_t xSize, uint32_t *pPhysicalAddr)
{
    // DMA要求：
    // 1. 物理连续
    // 2. 无Cache或Cache已维护
    // 3. 地址对齐
    
    void *ptr = pvPortMalloc(xSize + 32);  // 额外空间用于对齐
    if (ptr == NULL) return NULL;
    
    // 32字节对齐（Cache Line大小）
    uint32_t aligned = ((uint32_t)ptr + 31) & ~31;
    
    // 返回物理地址（简化，实际需要MMU转换）
    if (pPhysicalAddr != NULL) {
        *pPhysicalAddr = aligned;  // 假设物理=虚拟
    }
    
    // 关键：禁用该区域的Cache（需配合MPU配置）
    // SCB_InvalidateDCache_by_Addr((uint32_t *)aligned, xSize);
    
    return (void *)aligned;
}
```

### 3.3 链接脚本配置

```ld
/* STM32F7xx.ld - 异构内存分区 */

MEMORY
{
    /* 内部Flash */
    FLASH (rx)  : ORIGIN = 0x08000000, LENGTH = 1024K
    
    /* 内部SRAM - 分区使用 */
    ITCM_RAM (rx)  : ORIGIN = 0x00000000, LENGTH = 16K   /* 指令TCM */
    DTCM_RAM (rw)  : ORIGIN = 0x20000000, LENGTH = 64K   /* 数据TCM，最快 */
    SRAM1 (rw)     : ORIGIN = 0x20010000, LENGTH = 176K  /* 通用SRAM */
    SRAM2 (rw)     : ORIGIN = 0x2003C000, LENGTH = 16K   /* DMA专用 */
    
    /* 外部SDRAM */
    SDRAM (rw)     : ORIGIN = 0xC0000000, LENGTH = 8M    /* 外部大容量 */
}

/* 导出符号供C代码使用 */
__SRAM_START__ = ORIGIN(DTCM_RAM);
__SRAM_END__ = ORIGIN(DTCM_RAM) + LENGTH(DTCM_RAM);
__SDRAM_START__ = ORIGIN(SDRAM);
__SDRAM_END__ = ORIGIN(SDRAM) + LENGTH(SDRAM);
__DMA_SRAM_START__ = ORIGIN(SRAM2);

/* DMA缓冲区专用段 */
.dma_buffer (NOLOAD) :
{
    . = ALIGN(32);  /* Cache Line对齐 */
    __dma_buffer_start__ = .;
    *(.dma_buffer)
    __dma_buffer_end__ = .;
} > SRAM2
```

### 3.4 DMA缓冲区分配示例

```c
// 方式1：静态分配到专用段
__attribute__((section(".dma_buffer"))) 
static uint8_t dma_tx_buffer[2048];

// 方式2：动态分配DMA安全内存
void vDmaTransferExample(void)
{
    uint8_t *tx_buf = pvAllocDmaSafe(2048, NULL);
    
    // 填充数据
    memcpy(tx_buf, source_data, 2048);
    
    // Cache维护（如果该区域有Cache）
    SCB_CleanDCache_by_Addr((uint32_t *)tx_buf, 2048);
    
    // 启动DMA传输
    HAL_DMA_Start(&hdma_usart1_tx, (uint32_t)tx_buf, 
                  (uint32_t)&USART1->DR, 2048);
    
    // 等待完成
    // ...
    
    vPortFree(tx_buf);
}
```

## 四、避坑与边界条件

> [!danger] 致命陷阱
> 1. **跨区域释放**：如果手动管理多堆句柄，必须确保 `free()` 使用正确的堆句柄
> 2. **DMA缓冲区Cache问题**：SDRAM通常有Cache，DMA传输前必须 `Clean`，接收后必须 `Invalidate`
> 3. **地址对齐**：DMA要求地址和数据长度都对齐到传输宽度

> [!warning] 避坑指南
> - heap_5的区域表必须按地址升序排列
> - 不要在运行时动态添加内存区域（虽然API支持，但风险高）
> - 外部内存访问延迟可能影响实时性，关键中断处理代码不要放SDRAM

### 4.1 性能对比实测

```c
// 内存访问延迟测试
void vBenchmarkMemoryRegions(void)
{
    volatile uint32_t *ptr_sram = (uint32_t *)0x20000000;
    volatile uint32_t *ptr_sdram = (uint32_t *)0xC0000000;
    uint32_t start, elapsed;
    
    // SRAM写入测试
    start = DWT->CYCCNT;
    for (int i = 0; i < 1000; i++) {
        ptr_sram[i] = i;
    }
    elapsed = DWT->CYCCNT - start;
    printf("SRAM write: %u cycles\n", elapsed);
    
    // SDRAM写入测试
    start = DWT->CYCCNT;
    for (int i = 0; i < 1000; i++) {
        ptr_sdram[i] = i;
    }
    elapsed = DWT->CYCCNT - start;
    printf("SDRAM write: %u cycles\n", elapsed);
    
    // 典型结果：SDRAM比SRAM慢5-20倍
}
```
