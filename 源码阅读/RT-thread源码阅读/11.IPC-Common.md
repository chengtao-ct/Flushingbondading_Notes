### Mailbox


#### 概念

RT-Thread 操作系统的消息箱用于线程间的通信，其特点是开销低、效率高等。消息箱中的每条消息只能存储固定大小的 4 个字节数据（在 32 位处理系统中，指针的大小为 4 个字节，因此每条消息只能存储一个指针）。这种消息传递方式也被称为消息交换。如下图所示，某个线程或中断服务例程会将 4 字节的消息发送到消息箱中，随后一个或多个线程可以从消息箱中接收并处理这些消息。

在 ISR 中，可以安全地使用非阻塞模式的邮件发送功能。这对于线程、中断服务程序以及定时器来说，都是向其他线程发送消息的有效方式。不过，邮件的接收过程则可能是阻塞式的，这取决于邮件箱中是否有消息，以及接收消息时所设置的超时时间。如果邮件箱中没有消息，且超时时间不为 0，那么邮件接收过程将会被阻塞。在这种情况下，只有线程才能负责接收邮件。

- 假设你要发送的不是一句短字符串，而是一个 1MB 的传感器数据缓冲区。如果你用消息队列（Message Queue），系统需要把这 1MB 数据老老实实拷贝进队列，再拷贝出来，不仅极慢还会导致内存爆炸。 而在这段邮箱代码中，**传递的仅仅是 4 个字节的“内存地址”**。不管背后的字符串有多长，发信和收信永远只搬运这 4 个字节。这就是嵌入式中极具艺术感的**零拷贝通信**。

```c
struct rt_mailbox
{
    struct rt_ipc_object parent;

    rt_uint32_t* msg_pool;            /* the start address of the mailbox buffer */
    rt_uint16_t size;                 /* the size of the mailbox buffer     */

    rt_uint16_t entry;                 /* the number of messages in the mailbox     */
    rt_uint16_t in_offset, out_offset;    /* the entry and exit pointer of the mailbox buffer   */
    rt_list_t suspend_sender_thread;      /* send the suspend and wait queue of the thread */
};
typedef struct rt_mailbox* rt_mailbox_t;
```

##### `rt_mb_init` and `rt_mb_detach`为静态的逻辑

```c
#ifdef RT_USING_MAILBOX

rt_err_t rt_mb_init(rt_mailbox_t mb,
                    const char  *name,
                    void        *msgpool,
                    rt_size_t    size,
                    rt_uint8_t   flag)
{
    /* 1. 极度防御性编程：确保句柄不为空，且排队规则必须是 FIFO 或 PRIO */
    RT_ASSERT(mb != RT_NULL);
    RT_ASSERT((flag == RT_IPC_FLAG_FIFO) || (flag == RT_IPC_FLAG_PRIO));

    /* 2. 面向对象思想（继承）：初始化最顶层的 Object 基类
     * 将邮箱对象打上 RT_Object_Class_MailBox 标签，并挂载到内核对象管理器的链表中 */
    rt_object_init(&(mb->parent.parent), RT_Object_Class_MailBox, name);

    /* 3. 设置 IPC 父类的阻塞排队规则（按优先级唤醒，还是先来后到） */
    mb->parent.parent.flag = flag;

    /* 4. 初始化 IPC 父类：主要是初始化用于存放“等待信件的收件人(线程)”的双向链表 */
    _ipc_object_init(&(mb->parent));

    /* 5. 核心业务初始化：绑定开发者预先分配好的静态内存池 */
    mb->msg_pool   = (rt_ubase_t *)msgpool; // 指向实际装信的“物理柜子”
    mb->size       = (rt_uint16_t)size;     // 柜子的总容量（能装几封信）
    mb->entry      = 0;                     // 当前信箱里的信件数量（初始为0）
    mb->in_offset  = 0;                     // 环形缓冲区的写指针（投递口）
    mb->out_offset = 0;                     // 环形缓冲区的读指针（取信口）

    /* 6. 初始化邮箱特有的“发送者挂起队列”
     * 为什么需要这个？因为当邮箱塞满时，发信人也需要一个地方排队睡觉 */
    rt_list_init(&(mb->suspend_sender_thread));
    
    /* 7. 初始化多核 SMP / 中断并发保护的自旋锁 */
    rt_spin_lock_init(&(mb->spinlock));

    return RT_EOK;
}
RTM_EXPORT(rt_mb_init);


rt_err_t rt_mb_detach(rt_mailbox_t mb)
{
    rt_base_t level;

    /* 1. 严格的安全校验 */
    RT_ASSERT(mb != RT_NULL);
    // 确保它确实是个邮箱
    RT_ASSERT(rt_object_get_type(&mb->parent.parent) == RT_Object_Class_MailBox);
    // 确保它是静态分配的系统对象（动态分配的必须用 rt_mb_delete，否则会导致内存泄漏！）
    RT_ASSERT(rt_object_is_systemobject(&mb->parent.parent));

    /* 2. 关中断 + 获取自旋锁（进入严格的临界区，防止拆毁途中发生任务调度或中断打断） */
    level = rt_spin_lock_irqsave(&(mb->spinlock));
    
    /* 3. 暴力清场：唤醒所有与此邮箱相关的阻塞线程
     * 唤醒所有正在等信的收件人（塞给它们一个 RT_ERROR 错误码） */
    rt_susp_list_resume_all(&(mb->parent.suspend_thread), RT_ERROR);
    /* 唤醒所有因为邮箱满了而在排队的发送者（同样塞给它们错误码） */
    rt_susp_list_resume_all(&(mb->suspend_sender_thread), RT_ERROR);
    
    /* 4. 恢复中断 + 释放自旋锁 */
    rt_spin_unlock_irqrestore(&(mb->spinlock), level);

    /* 5. 将邮箱对象从 RT-Thread 的内核对象管理器链表中摘除（注销户口） */
    rt_object_detach(&(mb->parent.parent));

    return RT_EOK;
}
RTM_EXPORT(rt_mb_detach);
```


- **双挂起队列（Double Suspend Lists）的设计之美：** 普通的 IPC（比如信号量）通常只有一个“等资源”的阻塞队列。但邮箱不同，它具有**双向背压（Backpressure）机制**：
    
    1. 没信时，收件人要阻塞（借用父类的 `suspend_thread` 链表）。
        
    2. 信箱满了时，发送人也要阻塞（邮箱独有的 `suspend_sender_thread` 链表）。 所以 `detach` 时必须砍掉两条藤蔓，否则就会遗留死锁的僵尸线程。
        
- **为什么用 `rt_spin_lock_irqsave`？** 这里使用了最强级别的保护——**保存当前中断状态、关闭全局中断、然后获取自旋锁**。因为在唤醒所有线程（修改多条双向链表）的过程中，数据结构处于极度脆弱的中间态。如果此时发生硬件中断，且中断服务函数里恰好又调用了发邮件的 API，系统链表立刻就会崩溃。这就叫**保护临界区（Critical Section）**。


### 发送邮件的逻辑


#### `_rt_mb_send_wait` 发送邮件的底层封装逻辑

```c
static rt_err_t _rt_mb_send_wait(rt_mailbox_t mb,
                                 rt_ubase_t   value,
                                 rt_int32_t   timeout,
                                 int suspend_flag)
{
    struct rt_thread *thread;
    rt_base_t level;
    rt_uint32_t tick_delta;
    rt_err_t ret;

    /* 1. 基础参数校验：确保是个合法的邮箱对象 */
    RT_ASSERT(mb != RT_NULL);
    RT_ASSERT(rt_object_get_type(&mb->parent.parent) == RT_Object_Class_MailBox);

    /* 2. 上下文环境自检 (极其关键) 
     * 如果在中断服务函数(ISR)中调用此函数，timeout 必须为 0！
     * 因为硬件中断决不能阻塞等待，否则会导致整个系统卡死（调度器崩溃）。*/
    RT_DEBUG_SCHEDULER_AVAILABLE(timeout != 0);

    tick_delta = 0;
    thread = rt_thread_self(); // 获取当前正在发信的线程控制块(TCB)

    RT_OBJECT_HOOK_CALL(rt_object_put_hook, (&(mb->parent.parent)));

    /* 3. 进入临界区：关中断并获取自旋锁，保护邮箱内部状态不被打断 */
    level = rt_spin_lock_irqsave(&(mb->spinlock));

    /* 4. 非阻塞模式快速返回：如果信箱满了，且调用者是个急性子(timeout==0)，直接解锁走人 */
    if (mb->entry == mb->size && timeout == 0)
    {
        rt_spin_unlock_irqrestore(&(mb->spinlock), level);
        return -RT_EFULL;
    }

    /* 5. 核心阻塞逻辑：邮箱已满，准备排队睡觉 (注意：必须用 while 而不是 if，防止虚假唤醒) */
    while (mb->entry == mb->size)
    {
        /* 每次睡眠前，重置线程的错误码 */
        thread->error = -RT_EINTR;

        /* 二次确认：如果没有等待时间了，直接返回满错误 */
        if (timeout == 0)
        {
            rt_spin_unlock_irqrestore(&(mb->spinlock), level);
            return -RT_EFULL;
        }

        /* 6. 将当前线程挂起到邮箱专属的“发送者等待队列”中 */
        ret = rt_thread_suspend_to_list(thread, &(mb->suspend_sender_thread),
                                        mb->parent.parent.flag, suspend_flag);
        if (ret != RT_EOK)
        {
            rt_spin_unlock_irqrestore(&(mb->spinlock), level);
            return ret; // 挂起失败（比如被信号打断），直接返回
        }

        /* 7. 如果愿意等(timeout>0)，则启动线程自带的“闹钟”(定时器) */
        if (timeout > 0)
        {
            rt_tick_t timeout_tick = timeout;
            tick_delta = rt_tick_get(); // 记录睡觉前的时间戳

            rt_timer_control(&(thread->thread_timer), RT_TIMER_CTRL_SET_TIME, &timeout_tick);
            rt_timer_start(&(thread->thread_timer));
        }
        
        /* 8. 释放自旋锁，并交出 CPU 控制权！ (当前线程在此停滞，系统去跑别的线程) */
        rt_spin_unlock_irqrestore(&(mb->spinlock), level);
        rt_schedule(); // <--- 线程在这里睡着了，直到被别人唤醒才会执行下一句！

        /* 9. 线程苏醒！检查是因为什么醒来的？(别人腾出空位了？闹钟响了？还是被强杀信号唤醒了？) */
        if (thread->error != RT_EOK)
        {
            return thread->error; // 如果不是正常唤醒，直接返回错误码（比如 -RT_ETIMEOUT）
        }

        /* 正常唤醒，重新获取自旋锁，准备发信 */
        level = rt_spin_lock_irqsave(&(mb->spinlock));

        /* 10. 时间补偿机制：如果中途被叫醒发现还是没空位(虚假唤醒)，必须扣除刚才睡觉花掉的时间 */
        if (timeout > 0)
        {
            tick_delta = rt_tick_get() - tick_delta;
            timeout -= tick_delta;
            if (timeout < 0)
                timeout = 0;
        }
    } // 如果醒来发现 entry 还是等于 size，就继续 while 循环睡觉

    /* 11. 终于抢到了空位！开始投递信件 (操作环形缓冲区 Ring Buffer) */
    mb->msg_pool[mb->in_offset] = value; // 把信件塞进投递口
    ++ mb->in_offset;                    // 投递口向后移一格
    if (mb->in_offset >= mb->size)       // 如果到了数组末尾，绕回 0
        mb->in_offset = 0;

    if(mb->entry < RT_MB_ENTRY_MAX)
        mb->entry ++; // 信件总数 +1
    else
    {
        rt_spin_unlock_irqrestore(&(mb->spinlock), level);
        return -RT_EFULL; 
    }

    /* 12. 查收件人队列：有没有收件人因为没信而在挨饿排队？ */
    if (!rt_list_isempty(&mb->parent.suspend_thread))
    {
        /* 把排在第一位的收件人拉出挂起队列 */
        rt_susp_list_dequeue(&(mb->parent.suspend_thread), RT_EOK);

        rt_spin_unlock_irqrestore(&(mb->spinlock), level);

        /* 13. 立刻触发一次调度！
         * 为什么要调度？因为刚才唤醒的收件人，优先级可能比当前发信人高！
         * 实时系统必须保证高优先级任务立刻执行。 */
        rt_schedule(); 

        return RT_EOK;
    }
    
    rt_spin_unlock_irqrestore(&(mb->spinlock), level);
    return RT_EOK;
}
```

- **为什么等待资源必须要用 `while` 循环？（防虚假唤醒/惊群效应）** 假设线程 A 因为信箱满睡着了。此时消费者取走一封信，唤醒了 A。A 醒来准备塞信，但在它拿到锁之前，系统发生了一个外部硬件中断，中断服务程序顺手发了一封急件，把刚空出的格子填满了！此时 A 拿到锁，如果不做 `while` 重新判定，直接往下塞信，就会把中断发来的急件覆盖掉。**用 while 重新判定条件，是 RTOS 中防止资源抢占冲突的黄金法则。**
    
- **无处不在的 `rt_schedule()`（抢占式调度精髓）** 在这个函数里，`rt_schedule()` 出现了两次：
    
    - **第一次（睡眠时）**：我没资源了，我主动交出 CPU。
        
    - **第二次（唤醒别人时）**：我发了一封信，唤醒了等信的 B。**为什么这里立刻就要调度？** 因为 B 的优先级可能比我当前线程还高！既然 B 饿着肚子等的数据已经到位了，RT-Thread 作为一个“实时”操作系统，必须立刻交出 CPU 让高优先级的 B 去跑。
        
- **中断打断机制 (Suspend Flags)** 函数结尾的那三个 `RTM_EXPORT` 宏 (`_wait`, `_wait_interruptible`, `_wait_killable`)，实际上底层调用的都是同一个核心，只是 `suspend_flag` 不同。这决定了线程在睡觉时，如果收到操作系统的异步信号（Signal，类似 Linux 的 `kill -9`），能不能被提前叫醒并返回异常。

仔细看这行代码：`tick_delta = rt_tick_get() - tick_delta; timeout -= tick_delta;`。如果一个带有超时机制的线程，在等待途中被叫醒（醒了发现位置又被抢了，还得继续等），那么它**剩下的等待时间绝不能重新算，必须扣除掉刚才睡觉花掉的时间！** 这是开发高鲁棒性带超时功能的嵌入式组件必须掌握的 C 语言技巧。

##### 调用的上层函数：紧急发送

```c
rt_err_t rt_mb_urgent(rt_mailbox_t mb, rt_ubase_t value)
{
    rt_base_t level;

    /* 1. 基础参数校验：确保是个合法的邮箱对象 */
    RT_ASSERT(mb != RT_NULL);
    RT_ASSERT(rt_object_get_type(&mb->parent.parent) == RT_Object_Class_MailBox);

    /* 调用系统钩子函数（供调试、追踪或性能分析使用） */
    RT_OBJECT_HOOK_CALL(rt_object_put_hook, (&(mb->parent.parent)));

    /* 2. 进入临界区：关中断并获取自旋锁，保护多核/中断并发下的环形缓冲区 */
    level = rt_spin_lock_irqsave(&(mb->spinlock));

    /* 3. 邮箱已满判断 (注意：加急邮件是非阻塞的！)
     * 这里没有 while 循环，也没有挂起线程排队的操作。
     * 因为发送加急邮件的往往是极其紧急的硬件中断或高优先级任务，如果满了只能立刻报错返回，绝不能等！*/
    if (mb->entry == mb->size)
    {
        rt_spin_unlock_irqrestore(&(mb->spinlock), level);
        return -RT_EFULL;
    }

    /* 4. 【核心黑科技】：逆向操作读指针 (out_offset)，实现“无损插队” 
     * 普通发信是推进 in_offset (写指针)；
     * 但加急发信，是把 out_offset (读指针) 往回退一格，然后把信塞进这个退出来的格子里！*/
    if (mb->out_offset > 0)
    {
        mb->out_offset --; // 读游标往前退一格
    }
    else
    {
        /* 如果读游标已经在数组头部(0)了，就绕回到数组末尾，维持环形缓冲区的特性 */
        mb->out_offset = mb->size - 1; 
    }

    /* 5. 投递加急信件：直接放在读指针当前指向的位置 
     * 这样收件人下一次调用 recv 取信时，第一眼看到的绝对是这封加急信！*/
    mb->msg_pool[mb->out_offset] = value;

    /* 6. 信件总数 +1 */
    mb->entry ++;

    /* 7. 唤醒收件人逻辑 (与普通发信完全一致) */
    if (!rt_list_isempty(&mb->parent.suspend_thread))
    {
        /* 将因为没信而挂起挨饿的收件人唤醒出列 */
        rt_susp_list_dequeue(&(mb->parent.suspend_thread), RT_EOK);

        /* 释放自旋锁并开中断 */
        rt_spin_unlock_irqrestore(&(mb->spinlock), level);

        /* 8. 触发调度！让高优先级的收件人立刻拿到加急信并执行 */
        rt_schedule();

        return RT_EOK;
    }

    /* 如果没人等信，就正常释放自旋锁并开中断 */
    rt_spin_unlock_irqrestore(&(mb->spinlock), level);

    return RT_EOK;
}
RTM_EXPORT(rt_mb_urgent);
```




**插队魔法 (`out_offset` 的逆运算)：** 普通的邮箱是一个 Ring Buffer（环形缓冲区）：

- **写普通信 (`rt_mb_send`)**：把数据放在 `in_offset`（写游标），然后 `in_offset` 往后走一格。
    
- **取信 (`rt_mb_recv`)**：从 `out_offset`（读游标）拿数据，然后 `out_offset` 往后走一格。
    
- **写加急信 (`rt_mb_urgent`)**：它根本不碰 `in_offset`，而是直接把 **`out_offset` 往后退一格**，然后把信塞进退出来的这个格子里！如果 `out_offset` 已经是 0 了怎么办？退无可退，就绕回到数组末尾 (`size - 1`)。


**指针欺骗模式 (Pointer Manipulation Pattern)：** 当我们想要在队列头部“插队”插入数据时，新手的第一反应往往是：“我要把数组里所有的元素都往后挪一个位置，腾出 `[0]` 这个位置给新数据”。但这会带来 $O(n)$ 的巨大内存拷贝开销，在实时系统中是致命的。 RT-Thread 告诉我们：**永远不要移动数据，移动指针！** 通过巧妙地将“读指针”往反方向走一步，我们就零成本地实现了“插队”操作，时间复杂度始终为绝对的 $O(1)$。这种思维方式在编写任何对性能有极致要求的基础软件（如驱动、网络协议栈、数据库引擎）时，都非常值得学习和借鉴。



#### 收信逻辑端

##### `_rt_mb_recv`

```c
static rt_err_t _rt_mb_recv(rt_mailbox_t mb, rt_ubase_t *value, rt_int32_t timeout, int suspend_flag)
{
    struct rt_thread *thread;
    rt_base_t level;
    rt_uint32_t tick_delta;
    rt_err_t ret;

    /* 1. 基础参数校验：确保是个合法的邮箱对象 */
    RT_ASSERT(mb != RT_NULL);
    RT_ASSERT(rt_object_get_type(&mb->parent.parent) == RT_Object_Class_MailBox);

    /* 2. 上下文环境自检：如果在中断里调用接收函数，timeout 必须为 0！
     * 中断决不能阻塞等待，否则会导致整个系统卡死。*/
    RT_DEBUG_SCHEDULER_AVAILABLE(timeout != 0);

    tick_delta = 0;
    thread = rt_thread_self(); // 获取当前正在请求收信的线程控制块(TCB)

    /* 调用系统钩子函数（供调试或性能分析使用） */
    RT_OBJECT_HOOK_CALL(rt_object_trytake_hook, (&(mb->parent.parent)));

    /* 3. 进入临界区：关中断并获取自旋锁，保护邮箱内部状态 */
    level = rt_spin_lock_irqsave(&(mb->spinlock));

    /* 4. 非阻塞模式快速返回：如果信箱是空的，且调用者是个急性子(timeout==0)，直接解锁走人 */
    if (mb->entry == 0 && timeout == 0)
    {
        rt_spin_unlock_irqrestore(&(mb->spinlock), level);
        return -RT_ETIMEOUT;
    }

    /* 5. 核心阻塞逻辑：邮箱没信，准备排队睡觉 (注意：必须用 while 而不是 if，防止虚假唤醒) */
    while (mb->entry == 0)
    {
        /* 每次睡眠前，重置线程的错误码 */
        thread->error = -RT_EINTR;

        /* 二次确认：如果没有等待时间了，直接返回超时错误 */
        if (timeout == 0)
        {
            rt_spin_unlock_irqrestore(&(mb->spinlock), level);
            thread->error = -RT_ETIMEOUT;
            return -RT_ETIMEOUT;
        }

        /* 6. 将当前收件人挂起到邮箱父类(IPC对象)自带的“收件人等待队列”中 */
        ret = rt_thread_suspend_to_list(thread, &(mb->parent.suspend_thread),
                                        mb->parent.parent.flag, suspend_flag);
        if (ret != RT_EOK)
        {
            rt_spin_unlock_irqrestore(&(mb->spinlock), level);
            return ret; // 挂起失败（比如被强杀信号打断），直接返回
        }

        /* 7. 如果愿意等(timeout>0)，则启动线程自带的“闹钟”(定时器) */
        if (timeout > 0)
        {
            rt_tick_t timeout_tick = timeout;
            tick_delta = rt_tick_get(); // 记录睡觉前的时间戳

            rt_timer_control(&(thread->thread_timer), RT_TIMER_CTRL_SET_TIME, &timeout_tick);
            rt_timer_start(&(thread->thread_timer));
        }

        /* 8. 释放自旋锁，并交出 CPU 控制权！(线程在此停滞睡觉) */
        rt_spin_unlock_irqrestore(&(mb->spinlock), level);
        rt_schedule(); // <--- 收件人在这里睡着了，直到有人发信把它唤醒！

        /* 9. 线程苏醒！检查是因为什么醒来的？(有人发信了？闹钟响了？还是被异常唤醒了？) */
        if (thread->error != RT_EOK)
        {
            return thread->error; // 如果不是正常唤醒(如超时)，直接返回错误码
        }
        
        /* 重新获取自旋锁，准备取信 */
        level = rt_spin_lock_irqsave(&(mb->spinlock));

        /* 10. 时间补偿机制：如果中途被叫醒发现信又被别人抢了(虚假唤醒)，必须扣除刚才睡觉花掉的时间 */
        if (timeout > 0)
        {
            tick_delta = rt_tick_get() - tick_delta;
            timeout -= tick_delta;
            if (timeout < 0)
                timeout = 0;
        }
    } // 如果醒来发现 entry 还是 0，就利用剩下的 timeout 继续 while 循环睡觉

    /* 11. 终于有信了！开始取信 (操作环形缓冲区 Ring Buffer) */
    *value = mb->msg_pool[mb->out_offset]; // 从读游标(out_offset)处把信拿走

    /* 12. 读游标向后移一格 */
    ++ mb->out_offset;
    if (mb->out_offset >= mb->size)        // 如果到了数组末尾，绕回 0
        mb->out_offset = 0;

    /* 13. 信件总数 -1 */
    if(mb->entry > 0)
    {
        mb->entry --;
    }

    /* 14. 查发件人队列：有没有发信人因为信箱满了而在挨饿排队？ */
    if (!rt_list_isempty(&(mb->suspend_sender_thread)))
    {
        /* 因为刚才我取走了一封信，腾出了一个空位，所以立刻唤醒排在第一位的发信人！ */
        rt_susp_list_dequeue(&(mb->suspend_sender_thread), RT_EOK);

        rt_spin_unlock_irqrestore(&(mb->spinlock), level);
        RT_OBJECT_HOOK_CALL(rt_object_take_hook, (&(mb->parent.parent)));

        /* 15. 立刻触发一次调度！
         * 因为刚才唤醒的发信人，优先级可能比当前收信人高！必须保证高优先级任务立刻抢占执行。 */
        rt_schedule();

        return RT_EOK;
    }
    
    /* 如果没人等着发信，正常释放自旋锁并返回 */
    rt_spin_unlock_irqrestore(&(mb->spinlock), level);
    RT_OBJECT_HOOK_CALL(rt_object_take_hook, (&(mb->parent.parent)));

    return RT_EOK;
}
```


👉 **1. 一句话总结 (Core Purpose)** `_rt_mb_recv` 的核心作用是**安全地从环形缓冲区中取出一封邮件**；如果没信则将当前线程挂起排队，取信成功后，它还肩负着“通知（唤醒）之前因邮箱满而阻塞的发信人”的重要使命。

#### 控制台


##### `rt_mb_control`

`rt_mb_control` 是邮箱对象的**多功能控制接口**；当前它主要扮演“复位键（Reset）”的角色，用于在不释放内存的情况下，瞬间清空邮箱内的所有信件，并暴力遣散所有正在阻塞等待的线程。

```c
rt_err_t rt_mb_control(rt_mailbox_t mb, int cmd, void *arg)
{
    rt_base_t level;

    /* 告诉编译器 arg 参数暂时没用到，防止报 "unused variable" 警告 */
    RT_UNUSED(arg);

    /* 1. 基础参数校验：确保指针不为空，且对象类型确实是邮箱 */
    RT_ASSERT(mb != RT_NULL);
    RT_ASSERT(rt_object_get_type(&mb->parent.parent) == RT_Object_Class_MailBox);

    /* 2. 检查控制指令：目前 RT-Thread 的邮箱控制只支持“重置(RESET)”这一种指令 */
    if (cmd == RT_IPC_CMD_RESET)
    {
        /* 3. 进入临界区：关中断并获取自旋锁，防止在重置过程中被中断或多核打断 */
        level = rt_spin_lock_irqsave(&(mb->spinlock));

        /* 4. 暴力清场：强制唤醒所有因为等信或等空位而挂起的线程
         * rt_susp_list_resume_all 会把等待队列里的线程全部拉出来，并塞给它们一个 RT_ERROR。
         * 这样这些线程醒来后，就知道邮箱被强行重置了，从而去执行异常处理。
         */
        rt_susp_list_resume_all(&(mb->parent.suspend_thread), RT_ERROR); // 唤醒等信的收件人
        rt_susp_list_resume_all(&(mb->suspend_sender_thread), RT_ERROR); // 唤醒等空位的发件人

        /* 5. 核心重置逻辑：格式化环形缓冲区
         * 注意：这里并没有用 memset 去清空 msg_pool 里的实际内存数据！
         * 而是直接把信件总数(entry)、读游标(out_offset)和写游标(in_offset)全部清零。
         * 只要游标归零了，之前的数据就变成了逻辑上的“垃圾数据”，会被后来的新信件直接覆盖。
         */
        mb->entry      = 0;
        mb->in_offset  = 0;
        mb->out_offset = 0;

        /* 6. 退出临界区：释放自旋锁并开中断 */
        rt_spin_lock_irqrestore(&(mb->spinlock), level);

        /* 7. 触发一次调度！
         * 为什么要调度？因为刚才在第4步我们强制唤醒了一批线程。
         * 如果其中某个被唤醒的线程优先级比当前正在执行重置操作的线程还要高，
         * 系统必须立刻将 CPU 交给它！
         */
        rt_schedule();

        return RT_EOK;
    }

    /* 如果传入了不支持的 cmd 指令，返回错误 */
    return -RT_ERROR;
}
RTM_EXPORT(rt_mb_control);
```


**UNIX 哲学的经典 `ioctl` 模式：** 你会发现，这个函数的名字不叫 `rt_mb_reset`，而叫 `rt_mb_control`。它的入参留了一个 `int cmd` 和一个 `void *arg`。 为什么要这么设计？这是借鉴了 Linux/UNIX 系统中著名的 `ioctl`（Input/Output Control）设计模式。 作为底层框架，以后如果需要给邮箱增加新功能（比如：动态扩容、查询剩余空间、设置特殊的阈值报警等），**完全不需要去修改 API 函数的签名**，只需要增加新的 `cmd` 宏定义，并通过 `void *arg` 传递不同类型的数据结构即可。这种设计保证了操作系统对外 API 接口的**极度稳定性和超强可扩展性**，是你设计大型 C 语言软件框架时必学的神级模式

```c
#include <rtthread.h>

#define THREAD_PRIORITY      10
#define THREAD_TIMESLICE     5

/* 1. 静态分配：邮箱控制块与物理内存池 */
static struct rt_mailbox mb;
/* 邮箱的物理容量是 128 字节。在 32 位系统中，一个指针占 4 字节，所以这个邮箱能装 128/4 = 32 封信 */
static char mb_pool[128];

/* 2. 准备好要发送的“信件内容”。注意：这些字符串存在全局只读数据段 (RO-Data) */
static char mb_str1[] = "I'm a mail!";
static char mb_str2[] = "this is another mail!";
static char mb_str3[] = "over";

ALIGN(RT_ALIGN_SIZE)
static char thread1_stack[1024];
static struct rt_thread thread1;

/* Thread 1: 扮演“消费者/收件人”的角色 */
static void thread1_entry(void *parameter)
{
    char *str; // 准备一个字符指针，用来接住发件人发过来的“信件地址”

    while (1)
    {
        rt_kprintf("thread1: try to recv a mail\n");

        /* 【核心黑科技：指针的指针】
         * rt_mb_recv 第二个参数要求传一个 rt_uint32_t*。
         * 我们想修改 str 这个指针本身的值（让它指向 mb_str1），所以必须传 str 的地址（&str）。
         * 然后强制类型转换为 (rt_uint32_t *) 欺骗编译器。
         * RT_WAITING_FOREVER：如果没有信，死等，绝不空转耗费 CPU！
         */
        if (rt_mb_recv(&mb, (rt_uint32_t *)&str, RT_WAITING_FOREVER) == RT_EOK)
        {
            rt_kprintf("thread1: get a mail from mailbox, the content:%s\n", str);
            
            /* 如果收到的是结束信号 "over"，跳出循环准备下班 */
            if(!strcmp(str, mb_str3))
                break;

            /* 处理完一封信，休息 100ms，让出 CPU 给别的线程 */
            rt_thread_mdelay(100);
        }
    }
    
    /* 优雅退出：因为 mb 是静态全局变量（不在 Heap 上），所以只能用 detach 从内核链表中脱离，绝对不能用 delete！ */
    rt_mb_detach(&mb);
}

ALIGN(RT_ALIGN_SIZE)
static char thread2_stack[1024];
static struct rt_thread thread2;

/* Thread 2: 扮演“生产者/发件人”的角色 */
static void thread2_entry(void *parameter)
{
    rt_uint8_t count;

    count = 0;
    while (count < 10)
    {
        count ++;
        if (count & 0x1) // 奇数次
        {
            /* 【核心黑科技：地址转整数】
             * rt_mb_send 第二个参数要求传一个 32 位的整数 (rt_uint32_t)。
             * 我们直接把字符串 mb_str1 的内存地址拿出来 (&mb_str1)，
             * 强转成 32 位整数发过去。我们发的是“钥匙”，不是“箱子里的东西”。
             */
            rt_mb_send(&mb, (rt_uint32_t)&mb_str1);
        }
        else // 偶数次
        {
            rt_mb_send(&mb, (rt_uint32_t)&mb_str2);
        }

        /* 每发一封信，慢悠悠地休息 200ms */
        rt_thread_mdelay(200);
    }

    /* 发送第 11 封信，也是最后一封：“over” */

    rt_thread_init(&thread2, "thread2", thread2_entry, RT_NULL,
                   &thread2_stack[0], sizeof(thread2_stack),
                   THREAD_PRIORITY, THREAD_TIMESLICE);

    return 0;
}
```

### 消息队列


消息队列能够接收来自线程或 ISR 的、长度不固定的消息，并将这些消息缓存在自己的内存空间中。其他线程也可以从消息队列中读取相应的消息。当消息队列为空时，正在读取消息的线程将会被挂起。当有新消息到达时，被挂起的线程会被唤醒，以便接收并处理该消息。消息队列是一种异步通信方式。

如果说**邮箱（Mailbox）**是用来传递“纸条”（4字节指针）的，那么**消息队列**就是用来传递“包裹”（真实的数据块）的。因为要装真实的数据，所以它的内存切分逻辑比邮箱复杂得多。

我先在源码中加入底层的“硬核注释”，帮你理清最晦涩的**内存池切分与链表初始化**逻辑，然后再为你进行深度拆解。


在创建消息队列时，首先需要从对象管理器中分配一个消息队列对象，然后为该消息队列对象分配内存空间，从而形成一个空的消息链表。所需内存大小为：消息大小 + 消息头信息大小 × 消息队列中的消息数量。


消息队列适用于需要发送较长消息的场景，包括线程之间的消息交换，以及在中断服务例程中向线程发送消息（需要注意的是，中断服务例程本身无法接收消息）。以下内容将从发送消息和同步消息两个角度来介绍消息队列的用法。


在一般的系统设计中，如何发送同步消息是一个常见的问题。此时，可以根据具体情况选择相应的实现方式：可以采用“消息队列+信号量”或“消息箱”的方式来处理两个线程之间的通信。发送消息的线程会将消息放入消息队列中。消息发送完成后，发送方会等待接收消息的线程给出确认。其工作原理如下图所示

#### 创建和销毁的逻辑


##### `rt_mq_init`和 `rt_mq_detach`


`rt_mq_init` 的核心作用是将一块**静态连续的内存块**，像切豆腐一样切成一个个带有“链表头”的标准包裹，并用单向链表把它们串起来，从而建立起一个支持真实数据拷贝的消息队列；而 `rt_mq_detach` 负责在队列废弃时安全地遣散所有收发线程并注销对象。



在之前的学习中，你可能会发现**邮箱**用的是**环形缓冲区（Ring Buffer 数组 + in/out 游标）**。为什么**消息队列**突然改用**单向链表**了呢？

- **内存拷贝的代价：** 邮箱传递的只是 4 字节指针，移动游标极快。但消息队列需要拷贝任意大小的数据。如果队列满了需要等待，或者优先级高的消息需要插队（类似于邮箱的 urgent 加急邮件），用数组去挪动几十上百字节的数据代价太大。
    
- **链表的优势：** 消息队列维护了两个链表：一个是**空闲链表（Free List）**，一个是**就绪链表（Ready List）**。发送消息时，从 Free List 摘下一个空包裹，填入数据，然后挂到 Ready List 尾部；如果发送加急消息，只需直接挂到 Ready List 头部即可。全过程只有指针的修改，**没有任何多余的数据搬运，实现了 O(1) 复杂度的插队操作**！



- **预分配内存池架构（Memory Pool Pattern）：** 在 `for` 循环中，RT-Thread 演示了嵌入式底层最经典的“内存池”设计。严苛的工业/车规系统中极其厌恶 `malloc()`，因为会产生内存碎片。这种在初始化时**一次性将整块静态内存切分好**，并在运行时只在空闲链表和就绪链表之间做节点交换的设计，保证了内存申请和释放的时间是绝对确定（Deterministic）的，不会产生任何碎片。
    
- **指针步长运算陷阱：** 看这行代码：`(rt_uint8_t *)mq->msg_pool + temp * (...)`。 为什么一定要先强转成 `rt_uint8_t *`（单字节指针）再做加法？因为 `msg_pool` 传进来是 `void *`，在 C 语言中 `void *` 是不能直接进行加减运算的（编译器不知道它的步长）。强转为 `uint8_t *` 后，步长严格变为 1 字节，这样 `temp * 大小` 计算出来的物理地址偏移量才会是绝对正确的。这是写底层驱动时必会的安全写法。






```c
static rt_err_t _rt_mq_send_wait(rt_mq_t mq,
                                 const void *buffer,
                                 rt_size_t size,
                                 rt_int32_t prio,
                                 rt_int32_t timeout,
                                 int suspend_flag)
{
    rt_base_t level;
    struct rt_mq_message *msg;
    rt_uint32_t tick_delta;
    struct rt_thread *thread;
    rt_err_t ret;

    RT_UNUSED(prio); // 如果没开启优先级队列宏，防止编译器报警告

    /* 1. 基础参数校验：确保不是空指针，且发送的数据大小不能为0 */
    RT_ASSERT(mq != RT_NULL);
    RT_ASSERT(rt_object_get_type(&mq->parent.parent) == RT_Object_Class_MessageQueue);
    RT_ASSERT(buffer != RT_NULL);
    RT_ASSERT(size != 0);

    /* 2. 中断上下文自检：如果在中断里发消息，timeout 必须为 0！ */
    RT_DEBUG_SCHEDULER_AVAILABLE(timeout != 0);

    /* 3. 容量超限拦截：你要发的包裹比预设的最大包裹(msg_size)还大，直接拒收 */
    if (size > mq->msg_size)
        return -RT_ERROR;

    tick_delta = 0;
    thread = rt_thread_self();
    RT_OBJECT_HOOK_CALL(rt_object_put_hook, (&(mq->parent.parent)));

    /* 4. 进入临界区，准备从空闲链表(Free List)里抢一个空包裹 */
    level = rt_spin_lock_irqsave(&(mq->spinlock));

    /* 获取一个空闲的包裹节点 */
    msg = (struct rt_mq_message *)mq->msg_queue_free;
    
    /* 如果没有空闲包裹（队列满了），且是不等待模式，直接解锁走人 */
    if (msg == RT_NULL && timeout == 0)
    {
        rt_spin_unlock_irqrestore(&(mq->spinlock), level);
        return -RT_EFULL;
    }

    /* 5. 核心阻塞逻辑：队列已满，准备排队睡觉 (经典的 while 防虚假唤醒机制) */
    while ((msg = (struct rt_mq_message *)mq->msg_queue_free) == RT_NULL)
    {
        thread->error = -RT_EINTR;
        if (timeout == 0)
        {
            rt_spin_unlock_irqrestore(&(mq->spinlock), level);
            return -RT_EFULL;
        }

        /* 挂起当前发信人到 mq->suspend_sender_thread 发件人等待队列 */
        ret = rt_thread_suspend_to_list(thread, &(mq->suspend_sender_thread),
                                        mq->parent.parent.flag, suspend_flag);
        if (ret != RT_EOK)
        {
            rt_spin_unlock_irqrestore(&(mq->spinlock), level);
            return ret;
        }

        /* 启动超时闹钟 */
        if (timeout > 0)
        {
            rt_tick_t timeout_tick = timeout;
            tick_delta = rt_tick_get();
            rt_timer_control(&(thread->thread_timer), RT_TIMER_CTRL_SET_TIME, &timeout_tick);
            rt_timer_start(&(thread->thread_timer));
        }

        rt_spin_unlock_irqrestore(&(mq->spinlock), level);
        rt_schedule(); /* <--- 线程在这里睡着了，交出 CPU */

        /* 醒来后检查是被谁唤醒的（正常唤醒还是异常/超时） */
        if (thread->error != RT_EOK)
            return thread->error;
            
        level = rt_spin_lock_irqsave(&(mq->spinlock));

        /* 睡眠时间补偿，继续 while 循环抢包裹 */
        if (timeout > 0)
        {
            tick_delta = rt_tick_get() - tick_delta;
            timeout -= tick_delta;
            if (timeout < 0) timeout = 0;
        }
    }

    /* 6. 终于抢到了一个空包裹 msg！将它从空闲链表中摘除 */
    mq->msg_queue_free = msg->next;

    /* 7. 【极致优化：解锁拷贝！】
     * 注意这里：在拷贝真实数据前，先把自旋锁释放了！
     * 因为 rt_memcpy 可能会非常耗时，绝不能在此期间关闭全局中断。
     */
    rt_spin_unlock_irqrestore(&(mq->spinlock), level);

    msg->next = RT_NULL;
    ((struct rt_mq_message *)msg)->length = size; /* 记录这个包裹里实际装了多少字节 */
    
    /* 把用户数据拷贝到包裹的“数据区” (跳过消息头结构体) */
    rt_memcpy(GET_MESSAGEBYTE_ADDR(msg), buffer, size);

    /* 8. 数据装填完毕，重新上锁！准备把包裹挂到“就绪链表(Ready List)”上 */
    level = rt_spin_lock_irqsave(&(mq->spinlock));

#ifdef RT_USING_MESSAGEQUEUE_PRIORITY
    /* 9A. 开启了优先级特性：根据优先级(prio)把包裹插入到链表的合适位置 */
    msg->prio = prio;
    if (mq->msg_queue_head == RT_NULL)
        mq->msg_queue_head = msg;

    struct rt_mq_message *node, *prev_node = RT_NULL;
    /* 遍历就绪链表，找到第一个优先级比我低的节点，插在它前面 */
    for (node = mq->msg_queue_head; node != RT_NULL; node = node->next)
    {
        if (node->prio < msg->prio)
        {
            if (prev_node == RT_NULL) mq->msg_queue_head = msg; /* 插在最头 */
            else prev_node->next = msg;
            msg->next = node;
            break;
        }
        if (node->next == RT_NULL) /* 我是优先级最低的，只能插在末尾 */
        {
            if (node != msg) node->next = msg;
            mq->msg_queue_tail = msg;
            break;
        }
        prev_node = node;
    }
#else
    /* 9B. 普通 FIFO 模式：直接把包裹挂到就绪链表的尾巴上 */
    if (mq->msg_queue_tail != RT_NULL)
        ((struct rt_mq_message *)mq->msg_queue_tail)->next = msg;

    mq->msg_queue_tail = msg;
    if (mq->msg_queue_head == RT_NULL)
        mq->msg_queue_head = msg;
#endif

    /* 10. 信箱里已有信件数 +1 */
    if(mq->entry < RT_MQ_ENTRY_MAX)
        mq->entry ++;
    else
    {
        rt_spin_unlock_irqrestore(&(mq->spinlock), level);
        return -RT_EFULL; 
    }

    /* 11. 检查有没有收件人正在饿肚子排队？ */
    if (!rt_list_isempty(&mq->parent.suspend_thread))
    {
        /* 唤醒排在第一位的收件人 */
        rt_susp_list_dequeue(&(mq->parent.suspend_thread), RT_EOK);
        rt_spin_unlock_irqrestore(&(mq->spinlock), level);
        /* 立刻调度，让高优先级收件人去跑 */
        rt_schedule();
        return RT_EOK;
    }
    
    rt_spin_unlock_irqrestore(&(mq->spinlock), level);
    return RT_EOK;
}
```

```c
rt_err_t rt_mq_urgent(rt_mq_t mq, const void *buffer, rt_size_t size)
{
    rt_base_t level;
    struct rt_mq_message *msg;

    /* 1. 基础参数校验：严谨的入口防线 */
    RT_ASSERT(mq != RT_NULL);
    RT_ASSERT(rt_object_get_type(&mq->parent.parent) == RT_Object_Class_MessageQueue);
    RT_ASSERT(buffer != RT_NULL);
    RT_ASSERT(size != 0);

    /* 2. 物理限制检查：如果包裹超过了队列设定的最大消息长度，直接拒绝 */
    if (size > mq->msg_size)
        return -RT_ERROR;

    RT_OBJECT_HOOK_CALL(rt_object_put_hook, (&(mq->parent.parent)));

    /* 3. 第一阶段：锁住资源，从空闲池中抢盒子 */
    level = rt_spin_lock_irqsave(&(mq->spinlock));

    /* 从空闲链表(Free List)的头部拿一个空节点 */
    msg = (struct rt_mq_message *)mq->msg_queue_free;
    
    /* 如果 msg == RT_NULL，说明内存池里没空盒子了（队满了），直接报错，绝不挂起！
     * 加急发送是典型的“非阻塞”操作，因为高优先级事件等不起。 */
    if (msg == RT_NULL)
    {
        rt_spin_unlock_irqrestore(&(mq->spinlock), level);
        return -RT_EFULL;
    }
    
    /* 把拿到的盒子从空闲链表中摘除 */
    mq->msg_queue_free = msg->next;

    rt_spin_unlock_irqrestore(&(mq->spinlock), level);

    /* 4. 【核心拷贝】：在无锁状态下进行数据填充
     * 因为我们已经把 msg 这个节点彻底“私有化”了（它是从 Free List 摘下来的），
     * 此时别的线程动不了它，所以在这里拷贝是不需要加锁的，极大降低了中断延迟。 */
    ((struct rt_mq_message *)msg)->length = size;
    rt_memcpy(GET_MESSAGEBYTE_ADDR(msg), buffer, size);

    /* 5. 第二阶段：再次上锁，完成“加急插队” */
    level = rt_spin_lock_irqsave(&(mq->spinlock));

    /* 【灵魂逻辑】：直接插在队头！
     * 让 msg 的 next 指向当前的队头，然后把头指针更新为 msg。 */
    msg->next = (struct rt_mq_message *)mq->msg_queue_head;
    mq->msg_queue_head = msg;

    /* 如果队列之前是空的，那这一个包裹即是头也是尾 */
    if (mq->msg_queue_tail == RT_NULL)
        mq->msg_queue_tail = msg;

    /* 6. 更新队列计数器，并进行防溢出检查 */
    if(mq->entry < RT_MQ_ENTRY_MAX)
    {
        mq->entry ++;
    }
    else
    {
        rt_spin_unlock_irqrestore(&(mq->spinlock), level);
        return -RT_EFULL; /* 异常情况：溢出 */
    }

    /* 7. 唤醒收件人（与普通发信逻辑完全一致） */
    if (!rt_list_isempty(&mq->parent.suspend_thread))
    {
        rt_susp_list_dequeue(&(mq->parent.suspend_thread), RT_EOK);

        rt_spin_unlock_irqrestore(&(mq->spinlock), level);

        /* 立刻调度：既然是加急信，说明收件人一定在等，立刻抢占执行！ */
        rt_schedule();

        return RT_EOK;
    }

    rt_spin_unlock_irqrestore(&(mq->spinlock), level);

    return RT_EOK;
}
RTM_EXPORT(rt_mq_urgent);
```

```c
static rt_ssize_t _rt_mq_recv(rt_mq_t mq,
                              void *buffer,
                              rt_size_t size,
                              rt_int32_t *prio,
                              rt_int32_t timeout,
                              int suspend_flag)
{
    rt_base_t level;
    struct rt_mq_message *msg;
    rt_uint32_t tick_delta;
    struct rt_thread *thread;
    rt_err_t ret;
    rt_size_t len;

    RT_UNUSED(prio); // 若未开启优先级特性，消除编译警告

    /* 1. 严谨的参数校验 */
    RT_ASSERT(mq != RT_NULL);
    RT_ASSERT(rt_object_get_type(&mq->parent.parent) == RT_Object_Class_MessageQueue);
    RT_ASSERT(buffer != RT_NULL);
    RT_ASSERT(size != 0);

    /* 2. 上下文环境自检：禁止在中断里阻塞等待 */
    RT_DEBUG_SCHEDULER_AVAILABLE(timeout != 0);

    tick_delta = 0;
    thread = rt_thread_self();
    RT_OBJECT_HOOK_CALL(rt_object_trytake_hook, (&(mq->parent.parent)));

    /* 3. 进入临界区 */
    level = rt_spin_lock_irqsave(&(mq->spinlock));

    /* 4. 非阻塞模式快速返回：若队列为空且 timeout 为 0，直接超时退出 */
    if (mq->entry == 0 && timeout == 0)
    {
        rt_spin_unlock_irqrestore(&(mq->spinlock), level);
        return -RT_ETIMEOUT;
    }

    /* 5. 阻塞等待：若队列为空，挂起当前线程 */
    while (mq->entry == 0)
    {
        thread->error = -RT_EINTR;
        if (timeout == 0)
        {
            rt_spin_unlock_irqrestore(&(mq->spinlock), level);
            thread->error = -RT_ETIMEOUT;
            return -RT_ETIMEOUT;
        }

        /* 将线程挂入队列的收件人等待链表 (parent.suspend_thread) */
        ret = rt_thread_suspend_to_list(thread, &(mq->parent.suspend_thread),
                                        mq->parent.parent.flag, suspend_flag);
        if (ret != RT_EOK)
        {
            rt_spin_unlock_irqrestore(&(mq->spinlock), level);
            return ret;
        }

        /* 若有超时，启动线程计时器 */
        if (timeout > 0)
        {
            rt_tick_t timeout_tick = timeout;
            tick_delta = rt_tick_get();
            rt_timer_control(&(thread->thread_timer), RT_TIMER_CTRL_SET_TIME, &timeout_tick);
            rt_timer_start(&(thread->thread_timer));
        }

        rt_spin_unlock_irqrestore(&(mq->spinlock), level);
        rt_schedule(); /* <--- 线程在这里阻塞睡眠 */

        /* 醒来检查错误码 */
        if (thread->error != RT_EOK)
            return thread->error;

        level = rt_spin_lock_irqsave(&(mq->spinlock));
        
        /* 时间补偿逻辑：扣除已消耗的等待时间 */
        if (timeout > 0)
        {
            tick_delta = rt_tick_get() - tick_delta;
            timeout -= tick_delta;
            if (timeout < 0) timeout = 0;
        }
    }

    /* 6. 成功抢到消息：从就绪链表头部摘除包裹 */
    msg = (struct rt_mq_message *)mq->msg_queue_head;
    mq->msg_queue_head = msg->next;
    if (mq->msg_queue_tail == msg)
        mq->msg_queue_tail = RT_NULL;

    /* 消息库存减一 */
    if(mq->entry > 0) mq->entry --;

    rt_spin_unlock_irqrestore(&(mq->spinlock), level);

    /* 7. 数据拷贝：取出消息长度，并将内容拷贝给 buffer */
    len = ((struct rt_mq_message *)msg)->length;
    if (len > size) len = size; /* 若消息过长，只拷贝用户能接收的部分 */
    rt_memcpy(buffer, GET_MESSAGEBYTE_ADDR(msg), len);

#ifdef RT_USING_MESSAGEQUEUE_PRIORITY
    if (prio != RT_NULL) *prio = msg->prio;
#endif

    /* 8. 将取出的包裹放回空闲链表 (Free List) 以供复用 */
    level = rt_spin_lock_irqsave(&(mq->spinlock));
    msg->next = (struct rt_mq_message *)mq->msg_queue_free;
    mq->msg_queue_free = msg;

    /* 9. 唤醒因队列满而挂起的发送者 */
    if (!rt_list_isempty(&(mq->suspend_sender_thread)))
    {
        rt_susp_list_dequeue(&(mq->suspend_sender_thread), RT_EOK);
        rt_spin_unlock_irqrestore(&(mq->spinlock), level);
        RT_OBJECT_HOOK_CALL(rt_object_take_hook, (&(mq->parent.parent)));
        rt_schedule();
        return len;
    }

    rt_spin_unlock_irqrestore(&(mq->spinlock), level);
    RT_OBJECT_HOOK_CALL(rt_object_take_hook, (&(mq->parent.parent)));
    return len;
}
```

#### 控制台

`rt_mq_control`

```c
rt_err_t rt_mq_control(rt_mq_t mq, int cmd, void *arg)
{
    rt_base_t level;
    struct rt_mq_message *msg;

    /* 告诉编译器 arg 参数暂时未用到，防止编译警告 */
    RT_UNUSED(arg);

    /* 1. 参数校验：确保 mq 不为空，且确保该对象确实是消息队列类型 */
    RT_ASSERT(mq != RT_NULL);
    RT_ASSERT(rt_object_get_type(&mq->parent.parent) == RT_Object_Class_MessageQueue);

    /* 2. 检查控制指令：目前 RT-Thread 的消息队列仅支持 RESET 指令 */
    if (cmd == RT_IPC_CMD_RESET)
    {
        /* 3. 进入临界区：关中断并获取自旋锁，确保重置过程的原子性，防止被中断或多核打断 */
        level = rt_spin_lock_irqsave(&(mq->spinlock));

        /* 4. 强制唤醒所有因为此队列而阻塞的线程
         * 无论是等消息的收件人，还是等空位的发件人，全部唤醒！
         * 并塞给它们 RT_ERROR 错误码，告诉它们：“别等了，队列已经重置了”
         */
        rt_susp_list_resume_all(&mq->parent.suspend_thread, RT_ERROR);
        rt_susp_list_resume_all(&(mq->suspend_sender_thread), RT_ERROR);

        /* 5. 【核心逻辑】：清空已有的消息，并将所有消息节点归还给空闲链表 (Free List) */
        while (mq->msg_queue_head != RT_NULL)
        {
            /* 取出队列头部的消息节点 */
            msg = (struct rt_mq_message *)mq->msg_queue_head;

            /* 链表操作：摘除头部，指向下一个 */
            mq->msg_queue_head = msg->next;
            if (mq->msg_queue_tail == msg)
                mq->msg_queue_tail = RT_NULL;

            /* 链表操作：将这个节点插入到空闲链表(msg_queue_free)的头部 */
            msg->next = (struct rt_mq_message *)mq->msg_queue_free;
            mq->msg_queue_free = msg;
        }

        /* 6. 重置计数器：将已有的消息条数归零 */
        mq->entry = 0;

        /* 7. 退出临界区：释放自旋锁 */
        rt_spin_unlock_irqrestore(&(mq->spinlock), level);

        /* 8. 触发调度：因为刚才唤醒了一批线程，如果其中有高优先级线程，必须立刻交出 CPU！ */
        rt_schedule();

        return RT_EOK;
    }

    return -RT_ERROR;
}
RTM_EXPORT(rt_mq_control);
```
- **统一的 IO 控制模式 (`ioctl` 风格)：** 这是非常经典的 Linux/POSIX 风格的控制函数设计。入参 `int cmd` 和 `void *arg` 使得这个函数具有极强的**扩展性**。即使未来要给消息队列增加“查询当前空闲空间”、“动态设置阈值”等功能，也无需改动函数签名。
    
- **内存节点的“垃圾回收”：** 这是最精彩的部分。它并没有像 `free` 那样销毁内存，而是遍历了 `msg_queue_head` 链表，将所有原本被“占用”的 `rt_mq_message` 节点，重新回收到 `mq->msg_queue_free` 链表中。这意味着被复位的队列，所有内存资源瞬间回到了“可用”状态，实现了零碎块的即时高效复用。
    
- **调度器的即时干预 (`rt_schedule`)：** 当重置消息队列时，我们强行唤醒了所有正在等消息的线程。如果这些线程的优先级高于当前线程，系统必须在重置操作后立刻执行 `rt_schedule()`，确保高优先级任务的响应延迟降到最低。

```c
static rt_uint8_t msg_pool[2048];

    {

        if (rt_mq_recv(&mq, &buf, sizeof(buf), RT_WAITING_FOREVER) >= 0)

            if (result != RT_EOK)
            {
                rt_kprintf("rt_mq_send ERR\n");
            }
            rt_kprintf("thread2: send message - %c\n", buf);
        }
        buf++;
        cnt++;
        rt_thread_mdelay(5); // 生产者的节奏，每 5ms 发一次
    }
}

int msgq_sample(void)
{
    rt_err_t result;

    /* 初始化消息队列：
     * 参数说明：
     * 1. mq: 控制块
     * 2. "mqt": 名字
     * 3. &msg_pool[0]: 内存池起始地址
     * 4. 1: 每条消息长度（字节）
     * 5. sizeof(msg_pool): 内存池总大小
     * 6. RT_IPC_FLAG_FIFO: 先进先出策略
     */
    result = rt_mq_init(&mq, "mqt", &msg_pool[0], 1, sizeof(msg_pool), RT_IPC_FLAG_FIFO);

    if (result != RT_EOK)
    {
        rt_kprintf("init message queue failed.\n");
        return -1;
    }

    /* 启动线程 */
    rt_thread_init(&thread1, "thread1", thread1_entry, RT_NULL, &thread1_stack[0], sizeof(thread1_stack), 25, 5);
    rt_thread_startup(&thread1);

    rt_thread_init(&thread2, "thread2", thread2_entry, RT_NULL, &thread2_stack[0], sizeof(thread2_stack), 25, 5);
    rt_thread_startup(&thread2);

    return 0;
}
MSH_CMD_EXPORT(msgq_sample, msgq sample);

```

- **数据拷贝 (Copy-based IPC)：** 消息队列是真正的“数据拷贝”通道。`rt_memcpy` 在此起到了至关重要的作用。发送方将局部变量 `buf` 的内容复制进队列，接收方再复制出来。这保证了**数据传递的安全性**——即便发送方的局部变量销毁了，消息内容依然保存在队列里。
    
- **动态优先级与实时响应：**
    
    消息队列通过“数据节点链表”的设计，实现了在保持数据隔离的同时，能够处理非线性的数据顺序（插队）。在硬实时任务中，这种“加急处理”是确保故障报警信号能绕过常规任务、获得最高响应优先级的关键。
    
- **生产者-消费者模式：** 代码中生产者（Thread 2）频率高（5ms），消费者（Thread 1）频率低（50ms）。RT-Thread 的消息队列充当了缓冲区（Buffer）的角色，将突发的短时数据吞吐变为了平滑的处理过程，这正是解决“生产者过快，消费者过慢”问题的标准手段。

### 💡 4. 思考与启发 (Takeaway)

- **内存安全原则：** 永远记住，静态分配的内核对象（如示例中的 `struct rt_messagequeue mq` 和全局数组 `msg_pool`）不需要也不会调用 `free`。在嵌入式编程中，这种确定性的内存布局是避免内存碎片的最佳实践。
    
- **API 语义化：** 注意代码中使用的 `sizeof(buf)`。当消息队列的单条消息长度定义为 1 字节时，发送结构体或指针也是可以的，只要你对应的发送和接收 `size` 匹配即可。这种灵活性使得消息队列可以传递极其多样化的数据结构。
